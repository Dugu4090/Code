# â”€â”€ Preflight â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€
preflight() {
    log_step "Checking system requirements..."

    # OS check
    local os_name
    os_name=$(uname -s 2>/dev/null || echo "unknown")
    if [ "${os_name}" != "Linux" ]; then
        die "Colide Code currently supports Linux only. Detected: ${os_name}"
    fi
    log_ok "OS: Linux"

    # Architecture check
    local arch
    arch=$(uname -m 2>/dev/null || echo "unknown")
    case "${arch}" in
        x86_64|amd64)
            log_ok "Architecture: ${arch} (x86_64)"
            ;;
        aarch64|arm64)
            log_warn "Architecture: ${arch} â€” binary is built for x86_64"
            log_warn "It may not work on ARM. Continuing anyway..."
            ;;
        *)
            die "Unsupported architecture: ${arch}. Colide requires x86_64."
            ;;
    esac

    # curl or wget
    if command -v curl >/dev/null 2>&1; then
        DOWNLOADER="curl"
        log_ok "Downloader: curl"
    elif command -v wget >/dev/null 2>&1; then
        DOWNLOADER="wget"
        log_ok "Downloader: wget"
    else
        die "Neither curl nor wget found. Install one and retry."
    fi

    # tar
    if ! command -v tar >/dev/null 2>&1; then
        die "tar not found. Install it and retry."
    fi
    log_ok "tar available"

    # file (optional, for verification)
    if command -v file >/dev/null 2>&1; then
        HAS_FILE_CMD=true
    else
        HAS_FILE_CMD=false
    fi

    # Write permissions
    if [ -w "${INSTALL_DIR}" ]; then
        SUDO=""
        log_ok "Install dir writable: ${INSTALL_DIR}"
    elif command -v sudo >/dev/null 2>&1; then
        SUDO="sudo"
        log_ok "Install dir: ${INSTALL_DIR} (will use sudo)"
    else
        die "Cannot write to ${INSTALL_DIR} and sudo is not available.\n" \
            "Run as root or install sudo."
    fi

    echo
}

# â”€â”€ Download â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€
download_binary() {
    log_step "Downloading Colide Code..."
    log_info "${DOWNLOAD_URL}"

    TMP_DIR=$(mktemp -d 2>/dev/null || mktemp -d -t colide_install)
    local tarball="${TMP_DIR}/Colide.tar"

    if [ "${DOWNLOADER}" = "curl" ]; then
        if ! curl -fSL --progress-bar \
            --retry 3 \
            --retry-delay 2 \
            -o "${tarball}" \
            "${DOWNLOAD_URL}" 2>&1; then
            die "Download failed. Check your internet connection and the URL."
        fi
    else
        if ! wget -q --show-progress \
            --tries=3 \
            -O "${tarball}" \
            "${DOWNLOAD_URL}" 2>&1; then
            die "Download failed. Check your internet connection and the URL."
        fi
    fi

    if [ ! -f "${tarball}" ] || [ ! -s "${tarball}" ]; then
        die "Downloaded file is empty or missing."
    fi

    local size
    size=$(stat -c%s "${tarball}" 2>/dev/null || stat -f%z "${tarball}" 2>/dev/null || echo "?")
    log_ok "Downloaded (${size} bytes)"
    echo
}

# â”€â”€ Extract â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€
extract_binary() {
    log_step "Extracting binary..."

    cd "${TMP_DIR}"

    if ! tar xf Colide.tar 2>/dev/null; then
        die "Failed to extract archive. It may be corrupt."
    fi

    # Find the colide binary
    local binary_path=""
    if [ -f "${BINARY_NAME}" ]; then
        binary_path="${BINARY_NAME}"
    elif [ -f "Colide/${BINARY_NAME}" ]; then
        binary_path="Colide/${BINARY_NAME}"
    else
        # Search for it
        binary_path=$(find . -name "${BINARY_NAME}" -type f 2>/dev/null | head -1)
    fi

    if [ -z "${binary_path}" ] || [ ! -f "${binary_path}" ]; then
        log_err "Contents of archive:"
        ls -la "${TMP_DIR}/" 2>/dev/null || true
        die "Binary '${BINARY_NAME}' not found in the archive."
    fi

    # Move to a clean location
    if [ "${binary_path}" != "${BINARY_NAME}" ]; then
        mv "${binary_path}" "${TMP_DIR}/${BINARY_NAME}"
    fi

    chmod +x "${TMP_DIR}/${BINARY_NAME}"
    log_ok "Extracted: ${BINARY_NAME}"
    echo
}

# â”€â”€ Verify â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€
verify_binary() {
    log_step "Verifying binary..."

    local bin="${TMP_DIR}/${BINARY_NAME}"

    # Check it's executable
    if [ ! -x "${bin}" ]; then
        die "Binary is not executable."
    fi
    log_ok "Executable permissions set"

    # Check ELF header
    if [ "${HAS_FILE_CMD}" = true ]; then
        local ftype
        ftype=$(file "${bin}" 2>/dev/null || echo "unknown")
        if echo "${ftype}" | grep -qi 'ELF'; then
            log_ok "Valid ELF binary"
        else
            log_warn "Binary type: ${ftype}"
            log_warn "Expected ELF binary â€” proceeding anyway"
        fi
    fi

    # Check size is reasonable (> 1MB)
    local size
    size=$(stat -c%s "${bin}" 2>/dev/null || stat -f%z "${bin}" 2>/dev/null || echo 0)
    if [ "${size}" -lt 1000000 ] 2>/dev/null; then
        log_warn "Binary is small (${size} bytes) â€” may be incomplete"
    else
        local size_mb
        size_mb=$(echo "scale=1; ${size}/1048576" | bc 2>/dev/null || echo "?")
        log_ok "Size: ${size_mb} MB"
    fi

    echo
}

# â”€â”€ Install â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€
install_binary() {
    log_step "Installing to ${INSTALL_DIR}/${BINARY_NAME}..."

    local src="${TMP_DIR}/${BINARY_NAME}"
    local dst="${INSTALL_DIR}/${BINARY_NAME}"

    # Check if already installed
    if [ -f "${dst}" ]; then
        log_info "Existing installation found â€” replacing"
    fi

    if [ -n "${SUDO}" ]; then
        if ! ${SUDO} mv -f "${src}" "${dst}"; then
            die "Failed to install. sudo mv returned an error."
        fi
        ${SUDO} chmod +x "${dst}"
    else
        if ! mv -f "${src}" "${dst}"; then
            die "Failed to install. Cannot write to ${INSTALL_DIR}."
        fi
        chmod +x "${dst}"
    fi

    log_ok "Installed: ${dst}"
    echo
}

# â”€â”€ Verify install â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€
verify_install() {
    log_step "Verifying installation..."

    # Check it exists
    if [ ! -f "${INSTALL_DIR}/${BINARY_NAME}" ]; then
        die "Binary not found at ${INSTALL_DIR}/${BINARY_NAME} after install."
    fi

    # Check it's in PATH
    if command -v "${BINARY_NAME}" >/dev/null 2>&1; then
        local which_path
        which_path=$(command -v "${BINARY_NAME}")
        log_ok "Found in PATH: ${which_path}"
    else
        log_warn "'${BINARY_NAME}' not found in PATH"
        log_info "You may need to add ${INSTALL_DIR} to your PATH:"
        log_info "  export PATH=\"${INSTALL_DIR}:\$PATH\""
    fi

    echo
}
